var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 {Перфиксная форма инкремента увеличивает и переписывает значение а на единицу и возвращает новое значение}
d = b++; alert(d);           // 1 {Постфиксная форма инкремента увеличивает и переписывает значение b на единицу, но возвращает старое значение.
				   Данная строка может быть переписана следующим образом:
				   d = b;
                                   alert(d);
				   b++; 
                                  }
c = (2+ ++a); alert(c);      // 5 {В первом шаге значение а было переписано и равняется 2. Перфиксная форма увеличивает новое значение а и возращает новое значение, мы получаем 2+3=5, => a=3}
d = (2+ b++); alert(d);      // 4 {После второго шага значение b равно 2, значение b увеличивается на 1, но из-за постфиксной формы инкремента возвращает старое значение: 2+2 = 4, => b=3}
alert(a);                    // 3 {Значение a было перезаписано 2 раза и на этом шаге равняется 3}
alert(b);                    // 3 {Значение b было перезаписано 2 раза и равняется 3}
